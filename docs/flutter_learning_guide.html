第一部分：环境搭建

1.1 安装 Flutter SDK

Linux 安装步骤

  1. 下载 Flutter SDK： 
    cd ~/development
git clone https://github.com/flutter/flutter.git -b stable
  2. 添加到 PATH： 
    export PATH="$PATH:$HOME/development/flutter/bin"
    将上面这行添加到 ~/.bashrc 或 ~/.zshrc 中。
  3. 运行 Flutter Doctor 检查环境： 
    flutter doctor
  4. 根据提示安装缺少的依赖（如 Android SDK、Chrome 等）。

2.3 类与对象

// 基本类
class Person {
  String name;
  int age;

  // 构造函数
  Person(this.name, this.age);

  // 命名构造函数
  Person.unknown() : name = '未知', age = 0;

  // 方法
  void introduce() {
    print('我是$name，今年$age岁');
  }

  // Getter
  bool get isAdult => age >= 18;

  // 重写 toString
  @override
  String toString() => 'Person($name, $age)';
}

// 使用

print(person.isAdult);  // true

// 继承
class Student extends Person {
  String school;
  Student(String name, int age, this.school) : super(name, age);

  @override
  void introduce() {
    super.introduce();
    print('就读于$school');
  }
}

2.4 异步编程（async/await）

// Future - 代表一个异步操作的结果
Future fetchUserData() async {
  // 模拟网络请求
  await Future.delayed(Duration(seconds: 2));
  return '用户数据';
}

// 使用 async/await
void loadData() async {
  print('开始加载...');
  String data = await fetchUserData();  // 等待异步结果
  print('数据: $data');
  print('加载完成');
}

// Future.then 链式调用
fetchUserData()
  .then((data) => print('数据: $data'))
  .catchError((error) => print('错误: $error'));

// Stream - 代表一系列异步事件
Stream countStream() async* {
  for (int i = 1; i <= 5; i++) {
    await Future.delayed(Duration(seconds: 1));
    yield i;  // 产出一个值
  }
}

// 监听 Stream
countStream().listen((count) {
  print('计数: $count');
});

// try-catch 处理异步错误
Future safeLoadData() async {
  try {
    var data = await fetchUserData();
    print(data);
  } catch (e) {
    print('加载失败: $e');
  } finally {
    print('操作结束');
  }
}
核心概念：Flutter 中大量使用 Future 和 async/await 处理异步操作，如网络请求、数据库读写、文件操作等。掌握异步编程是 Flutter 开发的基础。

第三部分：Flutter 核心概念

3.1 一切皆 Widget

Flutter 中一切都是 Widget。Widget 是 Flutter UI 的基本构建块，类似于乐高积木。每个 Widget 都描述了它在给定配置和状态下应该如何显示。

// 一个最简单的 Flutter 应用
import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: '我的应用',
      home: Scaffold(
        appBar: AppBar(title: Text('首页')),
        body: Center(
          child: Text('你好，Flutter！'),
        ),
      ),
    );
  }
}

Widget 的两种类型

类型特点示例
StatelessWidget无状态，创建后不会改变Text、Icon、IconButton
StatefulWidget有状态，可以在运行时改变Checkbox、TextField、Slider
// StatelessWidget - 无状态组件
class Greeting extends StatelessWidget {
  final String name;

  const Greeting({Key? key, required this.name}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Text('你好, $name!');
  }
}

// StatefulWidget - 有状态组件
class Counter extends StatefulWidget {
  @override
  _CounterState createState() => _CounterState();
}

class _CounterState extends State {
  int _count = 0;

  void _increment() {
    setState(() {  // 调用 setState 触发 UI 刷新
      _count++;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('点击次数: $_count'),
        ElevatedButton(
          onPressed: _increment,
          child: Text('加 1'),
        ),
      ],
    );
  }
}
重要：setState() 会通知 Flutter 框架当前 State 已改变，需要重新调用 build() 方法更新 UI。不要在 build() 中调用 setState()，否则会死循环。

3.2 Widget 生命周期

StatefulWidget 的生命周期方法调用顺序：

  1. createState() - 创建 State 对象
  2. initState() - 初始化状态（只调用一次）
  3. didChangeDependencies() - 依赖变化时调用
  4. build() - 构建 UI（可能多次调用）
  5. didUpdateWidget() - Widget 配置变化时调用
  6. setState() - 触发重建
  7. deactivate() - 从树中移除时调用
  8. dispose() - 销毁时调用（释放资源）
class MyWidget extends StatefulWidget {
  @override
  _MyWidgetState createState() => _MyWidgetState();
}

class _MyWidgetState extends State {
  late TextEditingController _controller;

  @override
  void initState() {
    super.initState();
    _controller = TextEditingController();
    print('initState 被调用');
  }

  @override
  void didChangeDependencies() {
    super.didChangeDependencies();
    print('didChangeDependencies 被调用');
  }

  @override
  Widget build(BuildContext context) {
    return TextField(controller: _controller);
  }

  @override
  void dispose() {
    _controller.dispose();  // 必须释放资源！
    super.dispose();
  }
}

3.3 Key 的作用

Key 用于帮助 Flutter 识别 Widget，在列表中尤为重要：

// ValueKey - 基于值的 Key
ListView(
  children: items.map((item) =>
    ListTile(
      key: ValueKey(item.id),  // 帮助 Flutter 识别列表项
      title: Text(item.name),
    )
  ).toList(),
)

// Key 的类型：
// - ValueKey: 基于值
// - UniqueKey: 每次都是唯一的
// - ObjectKey: 基于对象引用

主轴对齐方式（MainAxisAlignment）

值效果
start靠左/顶部对齐
end靠右/底部对齐
center居中
spaceBetween两端对齐，中间等距
spaceAround每个元素两侧等距
spaceEvenly所有间距完全相等

6.2 Provider

// 1. 定义数据模型
class CounterModel extends ChangeNotifier {
  int _count = 0;
  int get count => _count;

  void increment() {
    _count++;
    notifyListeners();  // 通知监听者刷新
  }
}

// 2. 在顶层提供数据
void main() {
  runApp(
    ChangeNotifierProvider(
      create: (_) => CounterModel(),
      child: MyApp(),
    ),
  );
}

// 3. 在子 Widget 中使用
class CounterPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    // 读取数据
    final counter = context.watch();

    return Scaffold(
      body: Center(
        child: Text('${counter.count}', style: TextStyle(fontSize: 48)),
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: () => context.read().increment(),
        child: Icon(Icons.add),
      ),
    );
  }
}

// 只在部分 UI 刷新（优化性能）
class CountDisplay extends StatelessWidget {

    );

  }
}

6.3 BLoC（Business Logic Component）

// 本项目使用的状态管理方案

// 1. 定义 Event
abstract class CounterEvent {}
class IncrementEvent extends CounterEvent {}
class DecrementEvent extends CounterEvent {}

// 2. 定义 State
class CounterState {
  final int count;
  CounterState({this.count = 0});
}

// 3. 定义 BLoC
class CounterBloc extends Bloc {
  CounterBloc() : super(CounterState()) {
    on((event, emit) {
      emit(CounterState(count: state.count + 1));
    });
    on((event, emit) {
      emit(CounterState(count: state.count - 1));
    });
  }
}

// 4. 在 Widget 中使用
BlocProvider(
  create: (_) => CounterBloc(),
  child: BlocBuilder(
    builder: (context, state) {
      return Text('${state.count}');
    },
  ),
)

// 发送事件
context.read.add(IncrementEvent());
BLoC vs Provider：BLoC 适合复杂业务逻辑，事件驱动，便于测试；Provider 适合简单状态共享，使用更简洁。本项目采用 BLoC 模式。

// GET 请求
Future> fetchUsers() async {
  final response = await http.get(
    Uri.parse('https://api.example.com/users'),
    headers: {'Authorization': 'Bearer $token'},
  );

  if (response.statusCode == 200) {
    List data = json.decode(response.body);
    return data.map((json) => User.fromJson(json)).toList();
  } else {
    throw Exception('加载失败');
  }
}

// POST 请求
Future createUser(User user) async {
  final response = await http.post(
    Uri.parse('https://api.example.com/users'),
    headers: {'Content-Type': 'application/json'},
    body: json.encode(user.toJson()),
  );

  if (response.statusCode == 201) {
    return User.fromJson(json.decode(response.body));
  } else {
    throw Exception('创建失败');
  }
}

9.2 SQLite 数据库

// 使用 sqflite 包
import 'package:sqflite/sqflite.dart';
import 'package:path/path.dart';

class DatabaseHelper {
  static Database? _database;

  Future get database async {
    _database ??= await _initDatabase();
    return _database!;
  }

  Future _initDatabase() async {
    String path = join(await getDatabasesPath(), 'my_app.db');
    return openDatabase(
      path,
      version: 1,
      onCreate: (db, version) async {
        await db.execute('''
          CREATE TABLE notes (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            content TEXT,
            created_at TEXT DEFAULT CURRENT_TIMESTAMP
          )
        ''');
      },
    );
  }

  // CRUD 操作
  Future insertNote(Map note) async {
    final db = await database;
    return await db.insert('notes', note);
  }

  Future>> getNotes() async {
    final db = await database;
    return await db.query('notes', orderBy: 'created_at DESC');
  }

  Future updateNote(int id, Map note) async {
    final db = await database;
    return await db.update('notes', note, where: 'id = ?', whereArgs: [id]);
  }

  Future deleteNote(int id) async {
    final db = await database;
    return await db.delete('notes', where: 'id = ?', whereArgs: [id]);
  }
}

10.2 性能优化

// 1. 使用 const 构造函数
// 好的做法
const Text('不会改变的文字')
const EdgeInsets.all(16)
const Icon(Icons.star)

// 2. 避免在 build 中创建对象
// 不好
Widget build(BuildContext context) {
  final style = TextStyle(fontSize: 16);  // 每次 build 都创建
  return Text('Hello', style: style);
}

// 好（使用 const 或提为静态变量）
static const _style = TextStyle(fontSize: 16);

// 3. 使用 ListView.builder 而不是 ListView
// 不好（一次性创建所有子 Widget）
ListView(children: items.map((e) => ItemWidget(e)).toList())

// 好（懒加载，只创建可见的子 Widget）
ListView.builder(
  itemCount: items.length,
  itemBuilder: (context, index) => ItemWidget(items[index]),
)

// 4. 拆分 Widget 减少重建范围
// 不好 - 整个页面重建
class MyPage extends StatefulWidget {
  ...
}

// 好 - 只重建需要变化的部分
class MyPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        const Header(),                    // 不会重建
        const SizedBox(height: 20),        // 不会重建
        CounterSection(),                  // 只有这部分重建
      ],
    );
  }
}
